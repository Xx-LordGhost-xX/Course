/* */ = ניתן לכתוב הערות 

<style> = יוצרת פקודה לצבעים 

ID כוכבית לפני יוצרת 
 
class נקודה לפי יוצרת 

{} = יצרת אפשרות לכתיבה

link = קובץ חיצוני 

font-family: = קובע את משפחת הגופנים (או סדרת משפחות) לסגנון

font-size: = גודל הגופן 

font-weight: = מחיל כמות מסוימת או יחסית של הדגשה על הגופן

Font-variant = קובע את סגנון האותיות הקטנות בטקסט.

font-style = טקסט יהיה נטוי או אלכסוני

text-align = לגרום לטקסט להיות מיושר 

center; = טקסט באמצע

right; = טקסט מצד ימין

left; = טקסט מצד שמואל
 
justify; = הטקסט נותן יותר רווחים (כמו בעיתון)

text-decoration = מוריד קו תחתי ללינקים 

background: = צבע הרקע של הדף

body{} = מכליל את כל הבודי

padding: = (שוליים פנימים) מציין את כמות הרווח שבין תוכן הרכיב והגבול שלו

margin: = (שוליים חיצוניים) מציין את כמות הרווח בין גבול הרכיב 

margin-bottom = שוליים חיצוני תחתית

float = ציין את הצד שבו רכיבים אחרים זורמים מסביב לרכיב הצף

Clear = מציין את הצדדים של רכיב כלשהו שאינם מאפשרים רכיבי ציפה אחרים

Same For All = מגדיר מאפייני ריווח זהים לצד העליון, הימני, התחתון והשמאלי של רכיב שהריווח מוחל עליו

Width and Height = קובע את הרוחב והגובה של הרכיב
 
Type = מגדיר את מראה הסגנון של הגבול

body{margin:} = רווח פנימי של החלון (רק באדי)

solid = קו מסגרת ישר

dashed = קו מסגרת מקווקו

direction = דואג שנוכל להשתמש בשפות מימין לשמואל כגון עיברית וערבית

background-image: url(); =  קובע את תמונת הרקע לרכיב

background-size: = גודל התמונה

background-position: = מיקום תמונה

background-attachment: =  משאיר את התונה באותו מקום שגוללים 

background-repeat: = קובע אם וכיצד תמונת הרקע תחזור על עצמה

background-repeat: No Repeat = מציגה את התמונה פעם אחת בתחילת הרכיב

background-repeat: Repeat = מציגה את התמונה באריחים מאחורי הרכיב, אופקית או אנכית

background-repeat Repeat-x ו-Repeat-y = מציגות פס אופקי או אנכי של תמונות, בהתאמה. התמונות נחתכות כדי להתאים לגבולות הרכיב

background Attachment = קובע אם תמונת הרקע תהיה קבועה במיקומה המקורי או תזוז עם התוכן בשעת גלילת התוכן

background Position (X) ו-Background Position (Y) = מציין את המיקום ההתחלתי של תמונת הרקע יחסית לרכיב

background Color = קובע צבע רקע לרכיב. תכונת צבע הרקע נתמכת בשני הדפדפנים

Letter Spacing = מגדיל או מקטין את הרווח בין אותיות או תווים. להקטנת הרווח בין תווים

Text Indent = מציין את מרחק ההזחה של השורה הראשונה של הטקסט

Vertical Align = מציין את היישור האנכי של הרכיב שעליו מוחל הסגנון

Text Align = קובע כיצד יתבצע יישור טקסט בתוך הרכיב. התכונה

Whitespace = קובע כיצד יטופל רווח לבן בתוך הרכיב. בחר מבין שלוש אפשרויות

Display = מציין אם רכיב יוצג 

Word Spacing = קובע את הריווח בין מילים

position: = מיקום התמונה

top = המיקום התמונה תיהיה למעלה

left = מיקום התמונה יהיה שמואל

right = המיקום התמונה יהיה ימין

relative; = לקשר את התמונה לקנטנר

absolute; = בתוך הלוגו (תמונה ) שנרצה לקשר 

Line-height = מגדיר את גובה השורה שבה ממוקם הטקסט. הגדרה זו נקראת ריווח שורות.

Text-decoration = מוסיף קו תחתון, קו עליון או קו חוצה לטקסט, או גורם לטקסט להבהב. 

Text-transform = הופך את האות הראשונה בכל מילה לאות גדולה בבחירה, או מגדיר תצוגה של כל הטקסט על אותיות גדולות או אותיות קטנות

Color = קובע את צבע הטקסט

<heaber> = החלק העליון של האתר שחוזר על עצמו

<footer> = החלק התחתון של האתר שחוזר על עצמו

<nav> = תפריט ניווט - גם חוזר על עצמו אבל הלינקים שיהיו פבנים גוגל יסרוק אותם הרבה יותר

<aside> = מידע שהוא שולי שלא בשבילו המשתמתש נכנס - כגון פרסומות או הצעות למאמרים/מוצרים דומים

<figuer> -> <figcaption> = figcaption פיגר תגית שנעטוף בה תמונה וטקסט. הטקסט יופיע בתוך 

<article> -> <section> = מה שנמצא בתוך הארטקל זה מידע הראשי בדף שבשבילו המשתמש נכנס. סקשן יהיה בארטיקל ומחייב לפתוח תגית אייץ 2,3 או 4 ואז פיקסה. ומשתמש בו שנרצה לחלקה ארטקל למספר תתי נושאים

<video> -> audio> = MP4, MP3, WAM תגיות להוספת סיטונים או אודיו 

<main> = לא תגית מיוחדת בדרך כלל תיהיה בין ההדר לפוטר
