                                             ראש <head>

<style> = יוצרת פקודה לצבעים 

ID כוכבית לפני יוצרת 
 
class נקודה לפי יוצרת 

{} = יצרת אפשרות לכתיבה

link = קובץ חיצוני 

font-family: =  סוג גופן

font-size: = גודל הגופן 

font-weight: = אם הטקסט יהיה מוגדש או לא 

font-style = טקסט יהיה נתוי או לא

text-align = לגרום לטקסט להיות מיושר 

center; = טקסט באמצע

right; = טקסט מצד ימין

left; = טקסט מצד שמואל
 
justify; = הטקסט נותן יותר רווחים (כמו בעיתון)

text-decoration = מוריד קו תחתי ללינקים 

background: = צבע הרקע של הדף

body{} = מכליל את כל הבודי

padding: = שוליים פנימים 

margin: = שוליים חיצוניים  

margin-bottom = שוליים חיצוני תחתית

float = גורם לדיב להיות אחד ליד השני

body{margin:} = רווח פנימי של החלון (רק באדי)

solid = קו מסגרת ישר

dashed = קו מסגרת מקווקו

direction = דואג שנוכל להשתמש בשפות מימין לשמואל כגון עיברית וערבית

background-image: url(); = הוספת תמונה

background-size: = גודל התמונה

background-position: = מיקום תמונה

background-attachment: =  משאיר את התונה באותו מקום שגוללים 

background-repeat: no-repeat; = נותן את הגודל של התמונה המקורית

position: = מיקום התמונה

top = המיקום התמונה תיהיה למעלה

left = מיקום התמונה יהיה שמואל

right = המיקום התמונה יהיה ימין

relative; = לקשר את התמונה לקנטנר

absolute; = בתוך הלוגו (תמונה ) שנרצה לקשר 

                       בתוך המנוע חיפוש בעיקר 

<heaber> = החלק העליון של האתר שחוזר על עצמו

<footer> = החלק התחתון של האתר שחוזר על עצמו

<nav> = תפריט ניווט - גם חוזר על עצמו אבל הלינקים שיהיו פבנים גוגל יסרוק אותם הרבה יותר

<aside> = מידע שהוא שולי שלא בשבילו המשתמתש נכנס - כגון פרסומות או הצעות למאמרים/מוצרים דומים

<figuer> -> <figcaption> = figcaption פיגר תגית שנעטוף בה תמונה וטקסט. הטקסט יופיע בתוך 

<article> -> <section> = מה שנמצא בתוך הארטקל זה מידע הראשי בדף שבשבילו המשתמש נכנס. סקשן יהיה בארטיקל ומחייב לפתוח תגית אייץ 2,3 או 4 ואז פיקסה. ומשתמש בו שנרצה לחלקה ארטקל למספר תתי נושאים

<video> -> audio> = MP4, MP3, WAM תגיות להוספת סיטונים או אודיו 

<main> = לא תגית מיוחדת בדרך כלל תיהיה בין ההדר לפוטר
